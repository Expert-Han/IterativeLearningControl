\begin{exam}
	\label{ex:Unc}
	Uncertain Example
	\begin{align}
	A = \begin{pmatrix}
	-.5 + a & 0 \\ 0 & -.4-a
	\end{pmatrix}, 
	B = \begin{pmatrix}
	1 \\ 1
	\end{pmatrix}, 
	C = \begin{pmatrix}
	.1 & .2
	\end{pmatrix},
	D = 1. x_0 = \begin{pmatrix}
	0 \\ 0
	\end{pmatrix}, N = 200. 
	\end{align}
	$a$ is an uncertain parameter with nominal value $\hat{a} = 1.5$ and variability $a \in [1.3, \; 1.8]$.
